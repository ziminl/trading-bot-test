#!/usr/bin/env python3
import numpy as np
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, Dropout, BatchNormalization
from UtilityModule import add_debug_log

def build_market_regime_model(input_dim):
    model = Sequential()
    model.add(Dense(128, activation='relu', input_dim=input_dim))
    model.add(BatchNormalization())
    model.add(Dropout(0.3))
    model.add(Dense(64, activation='relu'))
    model.add(BatchNormalization())
    model.add(Dropout(0.3))
    model.add(Dense(32, activation='relu'))
    model.add(Dense(3, activation='softmax'))  # 0: bearish, 1: neutral, 2: bullish
    model.compile(optimizer='adam', loss='categorical_crossentropy', metrics=['accuracy'])
    add_debug_log("MarketRegime", "모델 생성", "시장 강세/약세 감지 모델 개선 버전 생성 완료", level="critical")
    return model

def train_market_regime_classifier(X_train, y_train, epochs=50, batch_size=32):
    input_dim = X_train.shape[1]
    model = build_market_regime_model(input_dim)
    model.fit(X_train, y_train, epochs=epochs, batch_size=batch_size, verbose=1)
    add_debug_log("MarketRegime", "학습", "시장 강세/약세 감지 모델 학습 완료", level="critical")
    return model

def predict_market_regime(model, state_vector):
    pred = model.predict(state_vector.reshape(1, -1))
    regime_idx = np.argmax(pred, axis=1)[0]
    regimes = {0: 'bearish', 1: 'neutral', 2: 'bullish'}
    regime = regimes.get(regime_idx, 'neutral')
    add_debug_log("MarketRegime", "예측", f"시장 상태 예측: {regime}")
    return regime
